from dataclasses import dataclass, field
from typing import Optional
@dataclass
class RatioResult:
value:        Optional[float]
status:       str         # COMPUTED|NOT_COMPUTABLE|BLOCKED|OUTLIER
reliability:  int         # 0-100 (درجة موثوقية)
source:       str  = ""   # المصدر: SEC|IFRS_ADAPTED|INFERRED
reason:       str  = ""   # سبب N/A إذا value=None
flag:         str  = ""   # تحذير خاص
display:      str  = ""   # النص المعروض للمستخدم
note:         str  = ""   # ملاحظة إضافية
anomaly_score:float = 0.0 # درجة الشذوذ 0-1
NA_MESSAGES = {
"ZERO_DENOMINATOR":     "مقام صفر — لا ينطبق",
"MISSING_SEC_CONCEPT":  "غير متاح في قاعدة البيانات",
"BLOCKED_BY_SECTOR":    "غير مطبَّق لهذا القطاع",
"IMPOSSIBLE_VALUE":     "قيمة مستحيلة — محذوفة",
"PRE_REVENUE_STAGE":    "شركة ناشئة — لا إيراد بعد",
"IFRS_NO_EQUIVALENT":   "لا مكافئ في IFRS",
"CURRENCY_MISMATCH":    "عملة غير موحَّدة",
}
def to_display(self, fmt="percent", lang="ar"):
if self.value is None:
msg = self.NA_MESSAGES.get(self.reason, self.reason or "غير متاح")
return f"— ({msg})"
if fmt == "percent":  return f"{self.value:.1%}"
if fmt == "multiple": return f"{self.value:.1f}×"
if fmt == "days":     return f"{self.value:.0f} يوم"
if fmt == "currency": return f"${self.value:,.0f}M"
return f"{self.value:.2f}"

5.2 محرك التعرف الشامل
from sentence_transformers import SentenceTransformer
import numpy as np
from sklearn.metrics.pairwise import cosine_similarity
import json, re
from pathlib import Path
class UniversalLabelEngine:
THRESHOLD  = 0.70
MODEL_NAME = "paraphrase-multilingual-MiniLM-L12-v2"  # متعدد اللغات
KB_FILE    = "data/label_knowledge.json"
def __init__(self):
self.model  = SentenceTransformer(self.MODEL_NAME)
self.corpus = UNIVERSAL_DICTIONARY
self.embs   = self._encode()
self.kb     = self._load_kb()
def _encode(self):
embs = {}
for canonical, cfg in self.corpus.items():
all_examples = []
for key, vals in cfg.items():
if isinstance(vals, list):
all_examples.extend(vals)
if all_examples:
e = self.model.encode(all_examples)
embs[canonical] = {"e":e,"mean":np.mean(e,axis=0)}
return embs
def recognize(self, raw, accounting_std="GAAP",
ticker=None, value=None, ctx=None):
# 1. ذاكرة متراكمة (الأسرع)
if raw in self.kb:
e = self.kb[raw]
if e.get("confidence",0) >= 0.88:
return {**e,"method":"learned"}
# 2. مطابقة دقيقة حسب المعيار المحاسبي
for c,cfg in self.corpus.items():
# تحقق من قيد القطاع
if cfg.get("bank_only") and "bank" not in (ctx or {}).get("sector",""):
continue
if cfg.get("insurance_only") and "insur" not in (ctx or {}).get("sector",""):
continue
if cfg.get("reit_only") and "reit" not in (ctx or {}).get("sector",""):
continue
# تحقق من المعيار المحاسبي ثم ALL
for std in [accounting_std, "ALL"]:
labels = cfg.get(std, [])
if isinstance(labels, list) and raw in labels:
r = {"canonical":c,"confidence":1.0,"method":"exact_std"}
self._learn(raw,r,ticker); return r
# 3. مطابقة مُطبَّعة
def norm(s): return re.sub(r"[\s\-_:.,/()（）]","",s.lower().strip())
rn = norm(raw)
for c,cfg in self.corpus.items():
for key,vals in cfg.items():
if not isinstance(vals,list): continue
for v in vals:
if rn == norm(v):
r = {"canonical":c,"confidence":0.95,"method":"normalized"}
self._learn(raw,r,ticker); return r
# 4. تشابه دلالي متعدد اللغات
raw_emb = self.model.encode([raw])[0]
best_c, best_s = None, 0.0
for c,data in self.embs.items():
ms = float(cosine_similarity([raw_emb],[data["mean"]])[0][0])
xs = float(np.max(cosine_similarity([raw_emb],data["e"])[0]))
score = 0.35*ms + 0.65*xs
if score > best_s: best_s,best_c = score,c
if best_s >= self.THRESHOLD:
r = {"canonical":best_c,"confidence":best_s,"method":"semantic_multilingual"}
self._learn(raw,r,ticker); return r
# 5. استدلال من القيمة والسياق
if value is not None and ctx:
r = self._value_inference(raw,value,ctx)
if r: self._learn(raw,r,ticker); return r
return {"canonical":None,"confidence":best_s,"method":"no_match","raw":raw}
def _value_inference(self, raw, value, ctx):
"""يستنتج البند من قيمته ومن القيم الأخرى"""
try:
v = float(value)
rev  = float(ctx.get("revenue",0) or 0)
cogs = float(ctx.get("cost_of_revenue",0) or 0)
ta   = float(ctx.get("total_assets",0) or 0)
# gross_profit = revenue - cogs
if rev and cogs and abs(v-(rev-cogs))/(rev-cogs+1) < 0.02:
return {"canonical":"gross_profit","confidence":0.88,
"method":"value_inference_gp"}
# total_assets ±3%
if ta and abs(v-ta)/ta < 0.03:
return {"canonical":"total_assets","confidence":0.85,
"method":"value_inference_ta"}
except: pass
return None
def _learn(self, raw, result, ticker=None):
self.kb[raw] = {**result,"ticker":ticker}
if len(self.kb) % 20 == 0: self._save_kb()
def _load_kb(self):
p = Path(self.KB_FILE)
return json.loads(p.read_text()) if p.exists() else {}
def _save_kb(self):
Path("data").mkdir(exist_ok=True)
Path(self.KB_FILE).write_text(
json.dumps(self.kb,ensure_ascii=False,indent=2))
def add_example(self, raw, canonical, std="ALL", source="manual"):
"""تصحيح يدوي — يُحسّن النموذج فوراً ويُعيد تشفيره"""
if canonical in self.corpus:
lst = self.corpus[canonical].setdefault(std,[])
if raw not in lst:
lst.append(raw)
all_ex = [v for k,vals in self.corpus[canonical].items()
if isinstance(vals,list) for v in vals]
e = self.model.encode(all_ex)
self.embs[canonical] = {"e":e,"mean":np.mean(e,axis=0)}
self.kb[raw] = {"canonical":canonical,"confidence":1.0,"source":source}
self._save_kb()

# ai/universal_sector_classifier.py
import numpy as np
from sklearn.ensemble import RandomForestClassifier, GradientBoostingClassifier
from sklearn.neural_network import MLPClassifier
from sklearn.preprocessing import StandardScaler
from sklearn.calibration import CalibratedClassifierCV  # معايرة الثقة
import joblib
from pathlib import Path
EXPLICIT_OVERRIDES = {
"INTC":"semiconductor_idm","TXN":"semiconductor_idm",
"MCHP":"semiconductor_idm","STM":"semiconductor_idm",
"TSM":"semiconductor_foundry","GFS":"semiconductor_foundry",
"BRK":"conglomerate","BRK.B":"conglomerate",
"DIB":"islamic_finance","EIB":"islamic_finance",
}
FEATURES = [
"gross_margin","operating_margin","net_margin","roic","roe","roa",
"revenue_cagr_3y","capex_to_revenue","rd_to_revenue",
"debt_to_equity","current_ratio","asset_turnover",
"fcf_yield","dividend_yield","interest_coverage",
"net_interest_margin","combined_ratio","ffo_yield",  # قطاعات خاصة
]
class UniversalSectorClassifier:
MODEL_PATH = "data/sector_model.joblib"
def __init__(self):
# Ensemble: RF + GBM + MLP → أدق من نموذج واحد
self.rf  = RandomForestClassifier(n_estimators=300,random_state=42)
self.gbm = GradientBoostingClassifier(n_estimators=200,random_state=42)
self.mlp = MLPClassifier(hidden_layer_sizes=(256,128,64),
activation="relu",max_iter=500,random_state=42)
# CalibratedClassifierCV: يُعايِر الثقة (Platt Scaling)
self.rf_cal  = CalibratedClassifierCV(self.rf,  method="isotonic")
self.gbm_cal = CalibratedClassifierCV(self.gbm, method="isotonic")
self.scaler  = StandardScaler()
self.trained = False
self.data    = []
self._load()
def features(self, ratios):
v = []
for f in FEATURES:
val = ratios.get(f)
try: v.append(float(val))
except: v.append(0.0)
return np.array(v)
def predict(self, ratios, ticker=None, profile=None):
# 0. استثناءات صريحة أولاً
if ticker in EXPLICIT_OVERRIDES:
s = EXPLICIT_OVERRIDES[ticker]
return {"sub_sector":s,"confidence":1.0,"method":"explicit"}
# 1. قواعد مختصة للكيانات غير التجارية
if profile:
r = self._special_entity_rules(ratios, profile)
if r: return r
# 2. نموذج قبل التدريب
if not self.trained:
return self._rules(ratios, profile)
X = self.scaler.transform(self.features(ratios).reshape(1,-1))
# Ensemble: وزن RF=40% GBM=40% MLP=20%
from sklearn.base import clone
SUBSECTORS = self.rf.classes_
rf_p  = dict(zip(SUBSECTORS,self.rf_cal.predict_proba(X)[0]))
gbm_p = dict(zip(SUBSECTORS,self.gbm_cal.predict_proba(X)[0]))
mlp_p = dict(zip(SUBSECTORS,self.mlp.predict_proba(X)[0]))
mp = {s:0.40*rf_p.get(s,0)+0.40*gbm_p.get(s,0)
+0.20*mlp_p.get(s,0) for s in SUBSECTORS}
best = max(mp, key=mp.get)
conf = mp[best]
# إذا الثقة منخفضة → أضف تحذيراً
if conf < 0.60:
return {"sub_sector":best,"confidence":conf,
"method":"ensemble_low_conf",
"top3":sorted(mp.items(),key=lambda x:-x[1])[:3],
"warning":"راجع التصنيف يدوياً"}
return {"sub_sector":best,"confidence":conf,
"method":"ensemble","all_probs":mp}
def _special_entity_rules(self, ratios, profile):
"""قواعد للكيانات غير التجارية وغير الربحية"""
country = getattr(profile,"country","US")
name    = getattr(profile,"name","").lower()
# شركة ناشئة بلا إيراد
if getattr(profile,"is_pre_revenue",False):
rd = ratios.get("rd_to_revenue",0) or 0
if rd > 0.5:
return {"sub_sector":"pre_revenue_biotech",
"confidence":0.80,"method":"pre_revenue_rule"}
return {"sub_sector":"pre_revenue_tech",
"confidence":0.80,"method":"pre_revenue_rule"}
# صندوق سيادي
if any(k in name for k in ["sovereign","fund","investment authority"]):
return {"sub_sector":"sovereign_fund",
"confidence":0.85,"method":"name_rule"}
# بنك إسلامي
if any(k in name for k in ["islamic","shariah","islami"]):
return {"sub_sector":"islamic_finance",
"confidence":0.85,"method":"name_rule"}
return None
def train(self, data):
"""يُدرّب النموذج الثلاثي"""
if len(data) < 20: return
X = np.array([self.features(r) for r,_ in data])
y = [l for _,l in data]
Xs = self.scaler.fit_transform(X)
self.rf.fit(Xs,y)
self.rf_cal.fit(Xs,y)
self.gbm.fit(Xs,y)
self.gbm_cal.fit(Xs,y)
self.mlp.fit(Xs,y)

joblib.dump({"rf":self.rf,"rf_cal":self.rf_cal,
"gbm":self.gbm,"gbm_cal":self.gbm_cal,
"mlp":self.mlp,"scaler":self.scaler,
"classes":self.rf.classes_,

7. المرحلة 5: Online Learning Engine — تعلم فوري
التعلم الدوري (كل 5 تصحيحات) يسبب تأخراً. هذا المكون يُحدّث النموذج فور كل تصحيح، ويكتشف انجراف البيانات تلقائياً.
# ai/online_learning_engine.py
from sklearn.linear_model import SGDClassifier  # يدعم partial_fit
from sklearn.preprocessing import StandardScaler
import numpy as np, json
from pathlib import Path
from collections import deque
class OnlineLearningEngine:
"""
تعلم فوري: كل تصحيح يدوي يُحدّث النموذج مباشرة.
لا انتظار لـ 5 تصحيحات — التحديث فوري.
"""
DRIFT_WINDOW  = 100   # نافذة كشف الانجراف
DRIFT_THRESH  = 0.15  # نسبة خطأ تُعتبر انجرافاً
def __init__(self):
# SGDClassifier يدعم partial_fit → تعلم فوري
self.model   = SGDClassifier(loss="log_loss",
random_state=42)
self.scaler  = StandardScaler()
self.fitted  = False
self.history = deque(maxlen=self.DRIFT_WINDOW)
self.drift_alerts = []
def update(self, ratios_vector, correct_sector,
ticker, classes_all):
"""
يُحدّث النموذج فور تصحيح واحد.
classes_all: كل القطاعات الممكنة (مطلوب لـ SGD).
"""
X = np.array(ratios_vector).reshape(1,-1)
if not self.fitted:
# أول تحديث: fit عادي
Xs = self.scaler.fit_transform(X)
self.model.partial_fit(Xs,[correct_sector],
classes=classes_all)
self.fitted = True
else:
Xs = self.scaler.transform(X)
pred = self.model.predict(Xs)[0]
# سجّل الخطأ لكشف الانجراف
self.history.append(1 if pred!=correct_sector else 0)
# تحديث فوري
self.model.partial_fit(Xs,[correct_sector])
self._check_drift(ticker)
def _check_drift(self, ticker):
"""يكتشف انجراف البيانات"""
if len(self.history) >= self.DRIFT_WINDOW:
error_rate = sum(self.history)/len(self.history)
if error_rate > self.DRIFT_THRESH:
alert = {
"type":   "DATA_DRIFT",
"ticker": ticker,
"error_rate": error_rate,
"msg": f"انجراف مكتشف: معدل خطأ {error_rate:.0%}",
"action": "أعد تدريب النموذج الكامل",
}
self.drift_alerts.append(alert)
def predict(self, ratios_vector):
if not self.fitted: return None
X = self.scaler.transform(
np.array(ratios_vector).reshape(1,-1))
pred  = self.model.predict(X)[0]
proba = self.model.predict_proba(X)[0]
return {"sector":pred,"confidence":float(max(proba))}
8. المرحلة 6: Anomaly Detection — كشف الشذوذات

# ai/anomaly_detector.py
from sklearn.ensemble import IsolationForest
import numpy as np
class AnomalyDetector:
"""
طبقتان للكشف:
1. قواعد صارمة: DSO=25000 يوم → مستحيل رياضياً
2. Isolation Forest: شذوذ إحصائي قياساً بالقطاع
"""
HARD_BOUNDS = {
"days_sales_outstanding":     (0,   365),
"days_inventory_outstanding": (0,   730),
"days_payable_outstanding":   (0,   730),
"ccc_days":                   (-365, 730),
"interest_coverage":          (-500, 1000),
"altman_z_score":             (-30,  80),
"net_debt_ebitda":            (-100, 100),
"gross_margin":               (-0.5, 1.0),
"operating_margin":           (-2.0, 1.0),
"roic":                       (-2.0, 5.0),
"revenue_cagr_3y":            (-1.0, 10.0),  # 1000%+ مشبوه
}
def __init__(self):
# نموذج منفصل لكل قطاع رئيسي
self.models = {}
def check_hard_bounds(self, metric, value, ticker, year, audit):
"""يحذف القيم المستحيلة رياضياً"""
if value is None: return None
try: v = float(value)
except: return None
bounds = self.HARD_BOUNDS.get(metric)
if not bounds: return v
lo, hi = bounds
if lo <= v <= hi: return v
audit.flag(year,f"IMPOSSIBLE_{metric.upper()}","CRITICAL",
f"{ticker} {year}: {metric}={v:.2f} خارج [{lo},{hi}]")
return None
def fit_sector_model(self, sub_sector, training_vectors):
"""يُدرّب نموذج Isolation Forest لكل قطاع"""
if len(training_vectors) < 10: return
X = np.array(training_vectors)
self.models[sub_sector] = IsolationForest(
contamination=0.05,  # نتوقع 5% شذوذات
random_state=42
).fit(X)
def score(self, sub_sector, ratio_vector):
"""
يُعيد درجة شذوذ 0-1.
0 = طبيعي جداً, 1 = شاذ جداً.
"""
model = self.models.get(sub_sector)
if not model: return 0.0
X = np.array(ratio_vector).reshape(1,-1)
# decision_function: سالب = شاذ
raw = model.decision_function(X)[0]
# تحويل لمقياس 0-1
return float(max(0, min(1, 0.5 - raw)))
def compute_fraud_indicators(self, ratios, prev_ratios):
"""
مؤشرات التلاعب المحاسبي (Beneish M-Score تبسيط):
- قفزة حادة في الذمم المدينة مقارنة بالإيراد
- تحسن غير طبيعي في هامش الإجمالي
- تدهور جودة الأرباح (Accruals مرتفعة)
"""
flags = []
try:
# DSI: تغير مخزون الأيام
dsi_curr = ratios.get("days_inventory_outstanding",0) or 0
dsi_prev = prev_ratios.get("days_inventory_outstanding",0) or 0
if dsi_prev > 0 and dsi_curr/dsi_prev > 1.5:
flags.append("INVENTORY_BUILDUP")
# DSRI: تغير أيام الذمم المدينة
dso_curr = ratios.get("days_sales_outstanding",0) or 0
dso_prev = prev_ratios.get("days_sales_outstanding",0) or 0
if dso_prev > 0 and dso_curr/dso_prev > 1.4:
flags.append("RECEIVABLES_SPIKE")
# GMI: تدهور الهامش الإجمالي
gm_curr = ratios.get("gross_margin",0) or 0
gm_prev = prev_ratios.get("gross_margin",0) or 0
if gm_prev > 0 and gm_curr/gm_prev < 0.8:
flags.append("MARGIN_DETERIORATION")
# Accruals مرتفعة
accruals = ratios.get("accruals_ratio",0) or 0
if abs(accruals) > 0.10:
flags.append("HIGH_ACCRUALS")
except: pass
return flags

def adapt(self, raw_data, accounting_std, audit, year):
"""
يُحوّل البيانات من أي معيار إلى التعريف الموحَّد.
يُوثّق كل تحويل في Audit Trail.
"""
mapping = self.MAPS.get(accounting_std, {})
adapted = {}
for label, value in raw_data.items():
canonical = mapping.get(label, label)
if canonical != label:
audit.correction(year, label, label, canonical,
f"adapted_{accounting_std}_to_canonical")
adapted[canonical] = value
# تعديلات IFRS 16: Lease liabilities → total_debt
if accounting_std == "IFRS":
lease = adapted.get("lease_liabilities", 0) or 0
if lease and "total_debt" in adapted:
original = adapted["total_debt"]
adapted["total_debt_pre_ifrs16"] = original
adapted["total_debt"] = float(original or 0) + float(lease)
audit.correction(year,"total_debt",original,
adapted["total_debt"],"ifrs16_lease_adjustment")
return adapted
def normalize_currency(self, data, from_currency,
to_currency="USD", rates=None):
"""يوحّد العملة إلى USD للمقارنة العالمية"""
if from_currency == to_currency: return data
if not rates: return data
rate = rates.get(from_currency, 1.0)
numeric_fields = [
"revenue","gross_profit","net_income",
"total_assets","total_equity","total_liabilities",
"total_debt","capital_expenditure",
]
normalized = dict(data)
for field in numeric_fields:
if field in normalized and normalized[field]:
try:
normalized[field] = float(normalized[field]) / rate
except: pass
return normalized
