# بعد حساب كل النسب لكل سنة:
def detect_and_null_duplicates(df, ticker):
    """إذا كانت قيمة 2025 = 2024 للأكثر من 5 نسب → الغِ 2025 كلها"""
    cols = [c for c in df.columns if str(c).isdigit()]
    if len(cols) < 2: return df
    yr_last, yr_prev = str(cols[-1]), str(cols[-2])
    numeric_rows = df.select_dtypes(include='number')
    matches = (numeric_rows[yr_last] == numeric_rows[yr_prev]).sum()
    if matches > 5:  # أكثر من 5 نسب متطابقة = بيانات مكررة
        print(f'WARNING: duplicate year detected')
        df[yr_last] = df[yr_last].where(df[yr_last] != df[yr_prev], other=float('nan'))
    return df

# قاموس جميع تجزئات الأسهم — يجب توسيعه لكل الشركات
STOCK_SPLITS = {
    'NVDA': [{'year': 2024, 'ratio': 10}],  # 10:1 يونيو 2024
    'TSLA': [{'year': 2022, 'ratio': 3}],   # 3:1 أغسطس 2022
    'AAPL': [{'year': 2020, 'ratio': 4}],   # 4:1 أغسطس 2020
}

def adjust_for_splits(ticker, year, shares_raw, ni, fair_value_total):
    splits = STOCK_SPLITS.get(ticker, [])
    cumulative_ratio = 1
    for split in splits:
        if int(year) < split['year']:
            cumulative_ratio *= split['ratio']
    shares_adjusted = shares_raw * cumulative_ratio
    eps_adjusted = ni / shares_adjusted
    fv_per_share = fair_value_total / shares_adjusted
    return shares_adjusted, eps_adjusted, fv_per_share

BETA_BOUNDS = {
    'technology':   (0.8, 3.0),
    'banking':      (0.8, 2.0),
    'healthcare':   (0.3, 1.5),
    'auto_ev':      (1.0, 3.5),
    'default':      (0.1, 3.0),
}

def validate_beta(ticker, year, beta_raw, sector, beta_prev_year):
    lo, hi = BETA_BOUNDS.get(sector, BETA_BOUNDS['default'])
    if not (lo <= beta_raw <= hi):
        print('WARNING: Beta out of range')
        # خيار 1: استخدم Beta السنة السابقة
        if beta_prev_year and (lo <= beta_prev_year <= hi):
            return beta_prev_year, 'fallback_prev_year'
        # خيار 2: استخدم وسط النطاق
        return (lo + hi) / 2, 'fallback_sector_median'
    return beta_raw, 'ok'

def detect_year_duplication(df, ticker, fc):
    """كشف إذا كانت أول سنة مكررة من الثانية"""
    cols = [c for c in df.columns[1:] if str(c).isdigit()]
    if len(cols) < 2: return
    yr0, yr1 = str(cols[0]), str(cols[1])
    numeric = df.select_dtypes(include='number')
    identical = (numeric[yr0] == numeric[yr1]).sum()
    total = numeric[yr0].notna().sum()
    if total > 0 and identical / total > 0.5:
        print(f'❌ {ticker}: سنة {yr0} = {yr1} ({{}{identical}/{{}{total}}} نسبة) → تكرار')
        df[yr0] = float('nan')  # امسح السنة المشبوهة

class CODEXValidator:
    BETA_MIN, BETA_MAX = 0.1, 3.5
    PE_MIN,  PE_MAX   = 3.0, 1500.0
    MARGIN_MIN = -0.5   # -50% أدنى هامش مقبول
    MARGIN_MAX =  0.95  # 95% أعلى هامش مقبول

    STOCK_SPLITS = {
        'NVDA': [('2024', 10)],
        'TSLA': [('2022', 3)],
        'AAPL': [('2020', 4)],
    }

    def validate(self, ticker, year, ratios: dict, prev_year_ratios: dict):
        errors, warnings = [], []

        # 1. كشف التكرار مع السنة السابقة
        if prev_year_ratios:
            matches = sum(1 for k in ratios
                if k in prev_year_ratios
                and ratios[k] == prev_year_ratios[k]
                and ratios[k] is not None)
            if matches > 5:
                errors.append('DUPLICATE_YEAR: نسب مكررة من السنة السابقة')

        # 2. Beta
        beta = ratios.get('beta')
        if beta and not (self.BETA_MIN <= beta <= self.BETA_MAX):
            errors.append('INVALID_BETA: Beta خارج النطاق المسموح')

        # 3. Stock Split
        splits = self.STOCK_SPLITS.get(ticker, [])
        for split_year, ratio in splits:
            if int(year) < int(split_year):
                eps = ratios.get('eps_basic', 0)
                eps_prev = (prev_year_ratios or {}).get('eps_basic', 0)
                if eps and eps_prev and eps / eps_prev > ratio * 0.8:
                    warnings.append('SPLIT_UNADJUSTED: EPS لم يُعدَّل للـ Stock Split')

        # 4. نسب مستحيلة
        for m in ['gross_margin','net_margin','operating_margin']:
            v = ratios.get(m)
            if v is not None and not (self.MARGIN_MIN <= v <= self.MARGIN_MAX):
                errors.append(f'IMPOSSIBLE_MARGIN: هامش مستحيل')

        return {'errors': errors, 'warnings': warnings,
                'status': 'FAIL' if errors else ('WARN' if warnings else 'OK')}
